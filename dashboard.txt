<?php
require_once 'config.php';
requireAdmin();

// Fetch basic stats
$totalOrders = $pdo->query("SELECT COUNT(*) FROM orders")->fetchColumn();
$totalRevenue = $pdo->query("SELECT SUM(total_price) FROM orders WHERE status = 'paid'")->fetchColumn() ?: 0;
$totalProducts = $pdo->query("SELECT COUNT(*) FROM products")->fetchColumn();
$totalCustomers = $pdo->query("SELECT COUNT(DISTINCT email) FROM orders")->fetchColumn();

// Fetch recent orders
$recentOrders = $pdo->query("SELECT * FROM orders ORDER BY created_at DESC LIMIT 5")->fetchAll();

// Fetch sales data for chart (last 7 days)
$salesData = $pdo->query("
    SELECT DATE(created_at) as date, SUM(total_price) as total 
    FROM orders 
    WHERE created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY) 
    GROUP BY DATE(created_at)
    ORDER BY date ASC
")->fetchAll();

header('Content-Type: application/json');
echo json_encode([
    'stats' => [
        'totalOrders' => $totalOrders,
        'totalRevenue' => $totalRevenue,
        'totalProducts' => $totalProducts,
        'totalCustomers' => $totalCustomers
    ],
    'recentOrders' => $recentOrders,
    'salesData' => $salesData
]);
?>
